
generic_debug_view = {
	show_wanted_divisions = {
		available = {
			
			always = no
		}
	}
	show_current_taxation = {
		available = {always = yes}
		complete_effect = {
			calculate_countrys_taxation = yes
		}
	}
	log_total_number_of_divisions = {
		available = {
			always = yes
		}	
		complete_effect = {	
			set_temp_variable = {
				var = d
				value = 0 
			}
			every_country = {
				add_to_temp_variable = {
					var = d
					value = THIS.num_divisions
				}
			}
			set_temp_variable = {
				var = e
				value = 0
			}
			every_country = {
				limit  = {
					is_in_earth_kingdom = yes
				}
				add_to_temp_variable = {
					var = e
					value = THIS.num_divisions
				}
			}
			log = "Total number of divisions: [?d], Fire Nation: [?FIR.num_divisions], Earth Kingdom: [?e]"
		}
	} 

	fir_effect = {
		allowed = {
			tag = FIR
		}
		available = {
			
		}

		complete_effect = {
			FIR_victory = yes
		}

	}
	avatars_location = {
		allowed = {
			always = yes
		}
		
	}
	stop_all_current_wars = {
		allowed = {
			always = yes
		}
		complete_effect = {
			every_country = {
				limit = {
					has_war = yes
				}
				every_possible_country = {
					limit = {
						has_war_with = PREV
					}
					white_peace = PREV
					give_military_access = PREV 
				}
			}
		}
	}
	declare_war_immediately = {
		allowed = {
			always = yes
		}
		target_trigger = {
			FROM = {
				NOT = {tag = ROOT} 
			}
		}
		complete_effect = {
			if = {
				limit = {
					FROM = {
						is_in_faction = yes
					}
				}
				FROM = {
					leave_faction = yes
				}
			}
			if = {
				limit = {
					ROOT = {
						is_in_faction = yes
					}
				}
				ROOT = {
					leave_faction = yes
				}
			}
			if = {
				limit = {
					FROM = {
						is_puppet = yes
					}
				}
				FROM.overlord = {
					release_puppet = FROM
				}
				FROM = {
					set_autonomy = {
						target = THIS
						autonomy_state = autonomy_free
					}
				}
			}
			if = {
				limit = {
					ROOT = {
						is_puppet = yes
					}
				}
				ROOT.overlord = {
					release_puppet = ROOT
				}
			}
			ROOT = {
				declare_war_on = {
					target = FROM
					type = annex_everything
				} 
			}
		}
	}
}


reoccupy_category = {

	show_genocided_territories = {

		icon = visible

		visible = {
			NOT = {
				has_country_flag = show_genocided_territories
			}
		}

		complete_effect = {
			set_country_flag = show_genocided_territories
		}

	}

	hide_genocided_territories = {

		icon = invisible

		visible = {
			has_country_flag = show_genocided_territories
		}

		complete_effect = {
			clr_country_flag = show_genocided_territories
		}

	}
	
	reinhabit_land = {

		icon = onmap_colonising


		allowed = {
			always = no
		}
		visible = {
			has_country_flag = show_genocided_territories
		}
		available = {
			NOT = {
				has_country_flag = establishing_a_colony
			}
			FROM = {
				NOT = {
					is_core_of = ROOT
					has_state_flag = recolonisation_in_progress
				}
			}
		}

		state_target = yes
		target_trigger = {
			
		}

		fire_only_once = no
		custom_cost_trigger = {
			check_variable = { var = money_variable compare = greater_than_or_equals value = 100 }
		}
		custom_cost_text = expenses_100

		days_remove = colony_growth@120
		#days_re_enable = 180
		on_map_mode = map_only
		complete_effect = {
			hidden_effect = { scoped_sound_effect  = "lose_gold" }
			subtract_from_variable = { var = money_variable value = 100 }
			
			FROM = {
				set_state_flag = recolonisation_in_progress
			}
			ROOT = {
				set_country_flag = establishing_a_colony
			}
		}
		cancel_effect = {
			FROM = {
				clr_state_flag = recolonisation_in_progress
			}
			ROOT = {
				clr_country_flag = establishing_a_colony
			}
		}

		modifier = {
			monthly_population = -0.1
			#stability_weekly_factor = -0.001
			expenses_per_pop = 1
		}

		remove_effect = {
			ROOT = {
				clr_country_flag = establishing_a_colony
				transfer_state = FROM
				add_stability = 0.1
			}
			FROM = {
				set_variable = {
					var = fire_pop
					value = ROOT.party_popularity_100@fire_culture
					tooltip = set_fire_pop_tt
				}
				set_variable = {
					var = earth_pop
					value = ROOT.party_popularity_100@earth_culture
					tooltip = set_earth_pop_tt
				}
				set_variable = {
					var = water_pop
					value = ROOT.party_popularity_100@water_culture
					tooltip = set_water_pop_tt
				}
				set_variable = {
					var = air_pop
					value = ROOT.party_popularity_100@air_culture
					tooltip = set_air_pop_tt
				}
				set_variable = {
					var = nonbender_pop
					value = ROOT.party_popularity_100@pluralist_culture
					tooltip = set_nonbender_pop_tt
				}

				clr_state_flag = recolonisation_in_progress
				add_dynamic_modifier = { modifier = new_settlement }
				force_update_dynamic_modifier = yes
			}
		}

		ai_will_do = {
			base = 0
		}
	}

}
