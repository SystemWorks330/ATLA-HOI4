	desc = firenation.5.d
	picture = GFX_report_event_dragon_attack

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_owned_state = {
				set_variable = { global.dragon_attack = THIS.id }
				set_state_flag = dragon_attack
			}
		}
	}

	option = {
		name = firenation.5.a

		every_state = {
			limit = {
				has_state_flag = dragon_attack
			}
			damage_all_building = yes

			set_temp_variable = { population_killed = state_population_k }
			multiply_temp_variable = { population_killed = -10 }
			
			add_manpower = var:population_killed

			clr_state_flag = dragon_attack
		}
	}
}
